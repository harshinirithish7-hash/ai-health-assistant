<script>
function predictDisease(){

    let symptom =
        document.getElementById("symptom")
        .value
        .toLowerCase();

    let prediction = "";

    if(symptom === "fever"){
        prediction = "Possible Viral Infection";
    }
    else if(symptom === "cough"){
        prediction = "Possible Respiratory Infection";
    }
    else if(symptom === "headache"){
        prediction = "Possible Migraine or Stress";
    }
    else if(symptom === "cold"){
        prediction = "Common Cold";
    }
    else if(symptom === "fatigue"){
        prediction = "Possible Weakness or Anemia";
    }
    else{
        prediction = "Please consult a doctor.";
    }

    document.getElementById("result").innerHTML =
        "AI Suggestion: " + prediction;
}
</script>
